def depositar(conta, valor, /):
    """Deposita um valor em uma conta espec√≠fica."""
    if valor > 0:
        conta["saldo"] += valor
        conta["extrato"] += f"Dep√≥sito: R$ {valor:.2f}\n"
        print("‚úÖ Dep√≥sito realizado com sucesso!")
    else:
        print("‚ùå Opera√ß√£o falhou! Valor inv√°lido.")
    return conta


def sacar(*, conta, valor, limite, limite_saques):
    """Saca um valor de uma conta espec√≠fica."""
    excedeu_saldo = valor > conta["saldo"]
    excedeu_limite = valor > limite
    excedeu_saques = conta["numero_saques"] >= limite_saques

    if excedeu_saldo:
        print("‚ùå Opera√ß√£o falhou! Saldo insuficiente.")
    elif excedeu_limite:
        print("‚ùå Opera√ß√£o falhou! Valor excede o limite.")
    elif excedeu_saques:
        print("‚ùå Opera√ß√£o falhou! N√∫mero m√°ximo de saques excedido.")
    elif valor > 0:
        conta["saldo"] -= valor
        conta["extrato"] += f"Saque: R$ {valor:.2f}\n"
        conta["numero_saques"] += 1
        print("‚úÖ Saque realizado com sucesso!")
    else:
        print("‚ùå Opera√ß√£o falhou! Valor inv√°lido.")
    return conta


def exibir_extrato(conta, /):
    """Mostra o extrato de uma conta espec√≠fica."""
    print("\n========== EXTRATO ==========")
    print("N√£o foram realizadas movimenta√ß√µes." if not conta["extrato"] else conta["extrato"])
    print(f"\nSaldo atual: R$ {conta['saldo']:.2f}")
    print("=============================")


def criar_usuario(usuarios):
    cpf = input("Informe o CPF (somente n√∫meros): ")
    usuario = filtrar_usuario(cpf, usuarios)
    if usuario:
        print("‚ö†Ô∏è J√° existe um usu√°rio com esse CPF.")
        return
    nome = input("Informe o nome completo: ")
    data_nascimento = input("Informe a data de nascimento (dd-mm-aaaa): ")
    endereco = input("Informe o endere√ßo (logradouro, n¬∫ - bairro - cidade / UF): ")

    usuarios.append({
        "nome": nome,
        "data_nascimento": data_nascimento,
        "cpf": cpf,
        "endereco": endereco
    })
    print("‚úÖ Usu√°rio criado com sucesso!")


def listar_usuarios(usuarios):
    """Exibe todos os usu√°rios cadastrados."""
    if not usuarios:
        print("‚ö†Ô∏è Nenhum usu√°rio cadastrado.")
        return

    print("\n========== USU√ÅRIOS ==========")
    for usuario in usuarios:
        print(f"Nome: {usuario['nome']}")
        print(f"CPF: {usuario['cpf']}")
        print(f"Data de Nascimento: {usuario['data_nascimento']}")
        print(f"Endere√ßo: {usuario['endereco']}")
        print("=" * 30)


def filtrar_usuario(cpf, usuarios):
    usuarios_filtrados = [usuario for usuario in usuarios if usuario["cpf"] == cpf]
    return usuarios_filtrados[0] if usuarios_filtrados else None


def criar_conta(agencia, numero_conta, usuarios):
    cpf = input("Informe o CPF do usu√°rio: ")
    usuario = filtrar_usuario(cpf, usuarios)
    if usuario:
        conta = {
            "agencia": agencia,
            "numero_conta": numero_conta,
            "usuario": usuario,
            "saldo": 0.0,
            "extrato": "",
            "numero_saques": 0
        }
        print(f"‚úÖ Conta {numero_conta} criada com sucesso para {usuario['nome']}!")
        return conta
    print("‚ùå Usu√°rio n√£o encontrado. Cria√ß√£o de conta encerrada.")


def listar_contas(contas):
    if not contas:
        print("‚ö†Ô∏è Nenhuma conta cadastrada.")
        return

    for conta in contas:
        print("=" * 40)
        print(f"Ag√™ncia: {conta['agencia']}")
        print(f"C/C: {conta['numero_conta']}")
        print(f"Titular: {conta['usuario']['nome']}")
        print(f"Saldo: R$ {conta['saldo']:.2f}")


def selecionar_conta(contas):
    if not contas:
        print("‚ö†Ô∏è Nenhuma conta dispon√≠vel.")
        return None

    listar_contas(contas)
    try:
        numero = int(input("Digite o n√∫mero da conta: "))
    except ValueError:
        print("‚ùå N√∫mero inv√°lido.")
        return None

    for conta in contas:
        if conta["numero_conta"] == numero:
            return conta

    print("‚ùå Conta n√£o encontrada.")
    return None


def main():
    LIMITE_SAQUES = 3
    LIMITE_VALOR = 500
    AGENCIA = "0001"

    usuarios = []
    contas = []

    menu = """
========= MENU =========
[d] Depositar
[s] Sacar
[e] Extrato
[u] Novo usu√°rio
[lu] Listar usu√°rios
[c] Nova conta
[l] Listar contas
[q] Sair
========================
=> """

    while True:
        opcao = input(menu).lower()

        if opcao == "u":
            criar_usuario(usuarios)

        elif opcao == "lu":
            listar_usuarios(usuarios)

        elif opcao == "c":
            numero_conta = len(contas) + 1
            conta = criar_conta(AGENCIA, numero_conta, usuarios)
            if conta:
                contas.append(conta)

        elif opcao == "l":
            listar_contas(contas)

        elif opcao in ("d", "s", "e"):
            conta = selecionar_conta(contas)
            if not conta:
                continue

            if opcao == "d":
                valor = float(input("Valor do dep√≥sito: "))
                depositar(conta, valor)

            elif opcao == "s":
                valor = float(input("Valor do saque: "))
                sacar(
                    conta=conta,
                    valor=valor,
                    limite=LIMITE_VALOR,
                    limite_saques=LIMITE_SAQUES
                )

            elif opcao == "e":
                exibir_extrato(conta)

        elif opcao == "q":
            print("üëã Saindo... Obrigado por usar o sistema banc√°rio!")
            break

        else:
            print("‚ùå Op√ß√£o inv√°lida. Tente novamente.")


if __name__ == "__main__":
    main()
